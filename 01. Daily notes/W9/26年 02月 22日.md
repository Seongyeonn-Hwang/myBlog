// ====== 설정 ======
const DAILY_FOLDER = "01. Daily notes"; // 일지 폴더 경로(필요 없으면 ""로 두고 아래 page query만 수정)
const SECTION_HEADING = "Daily notes."; // "### Daily notes." 에서 ### 제외한 텍스트
// ==================

function parseYYMD(s) {
  // "26年 02月 22日" 또는 "26年02月22日"
  const m = String(s ?? "").match(/(\d{2})年\s*(\d{1,2})月\s*(\d{1,2})日/);
  if (!m) return null;
  const yy = 2000 + Number(m[1]);
  const mm = Number(m[2]);
  const dd = Number(m[3]);
  return dv.luxon.DateTime.fromObject({ year: yy, month: mm, day: dd });
}

function pageDate(p) {
  // frontmatter date 우선, 없으면 파일명에서 파싱
  return parseYYMD(p.date) || parseYYMD(p.file.name);
}

async function extractSectionText(p, headingText) {
  // 파일 전체 텍스트를 읽고, "### headingText" 섹션만 잘라냄
  const full = await dv.io.load(p.file.path);
  const lines = full.split("\n");

  const h = "### " + headingText;
  let start = lines.findIndex(l => l.trim() === h);
  if (start === -1) return null;

  // 다음 같은 레벨 이상의 헤딩(###, ##, #) 나오기 전까지가 섹션
  let end = start + 1;
  while (end < lines.length) {
    const t = lines[end].trim();
    if (/^#{1,3}\s+/.test(t) && t.startsWith("### ") && t !== h) break; // 다음 ### 헤딩
    if (/^#{1,2}\s+/.test(t)) break; // 상위 헤딩 나오면 종료
    end++;
  }

  // 섹션 본문(헤딩 줄 제외)
  const body = lines.slice(start + 1, end).join("\n").trim();
  return body.length ? body : null;
}

function renderBlock(title, dateObj, content, fileLink) {
  dv.header(4, title);
  if (!dateObj || !content) {
    dv.paragraph("- (내용 없음)");
    return;
  }
  dv.paragraph(`${fileLink}`); //   \n${dateObj.toFormat("yyyy-LL-dd")}
  dv.paragraph(content);
}

// 오늘 페이지 기준 날짜
const today = pageDate(dv.current());
if (!today) {
  dv.paragraph("현재 노트의 date(YY年 MM月 DD日) 또는 파일명에서 날짜를 파싱하지 못했습니다.");
  return;
}

// 대상 날짜 계산
const targets = [
  { label: "일주일 전", dt: today.minus({ days: 7 }) },
  { label: "한달 전", dt: today.minus({ months: 1 }) },
  { label: "6개월 전", dt: today.minus({ months: 6 }) },
  { label: "1년 전",   dt: today.minus({ years: 1 }) },
];

// 일지 페이지 후보 목록
const pages = dv.pages(DAILY_FOLDER ? `"${DAILY_FOLDER}"` : "");

// 날짜로 페이지 찾기
function findPageByDate(dt) {
  return pages.where(p => {
    const d = pageDate(p);
    return d && d.hasSame(dt, "day");
  }).first();
}

dv.header(3, "");

let first = true;  
  
for (const t of targets) {  
  
const p = findPageByDate(t.dt);  
  
if (!first) dv.header(1, "---");  
first = false;  
  
if (!p) {  
dv.header(4, t.label);  
dv.paragraph("- (해당 날짜 노트 없음)");  
continue;  
}  
  
const section = await extractSectionText(p, SECTION_HEADING);  
renderBlock(t.label, t.dt, section, dv.fileLink(p.file.path));  
}
