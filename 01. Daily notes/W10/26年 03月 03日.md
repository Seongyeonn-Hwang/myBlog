```dataviewjs
try {  
// ====== 설정 ======  
const DAILY_FOLDER = "01. Daily notes";  
const SECTION_HEADING = "Daily notes.";  
// ==================  
  
function norm(s) {  
return String(s ?? "")  
.replace(/\u3000/g, " ")  
.replace(/[\u200B-\u200D\uFEFF]/g, "")  
.trim();  
}  
  
function parseYYMD(s) {  
const str = norm(s);  
const m = str.match(/(\d{2})年\s*(\d{1,2})月\s*(\d{1,2})日/);  
if (!m) return null;  
  
const yy = 2000 + Number(m[1]);  
const mm = Number(m[2]);  
const dd = Number(m[3]);  
  
return dv.luxon.DateTime.local(yy, mm, dd).startOf("day");  
}  
  
function pageDate(p) {  
return parseYYMD(p.date) || parseYYMD(p.file.name);  
}  
  
async function extractSectionText(p, headingText) {  
const full = await dv.io.load(p.file.path);  
const lines = full.split("\n");  
  
const h = "### " + headingText;  
let start = lines.findIndex(l => norm(l) === h);  
if (start === -1) return null;  
  
let end = start + 1;  
while (  
end < lines.length &&  
!/^#{1,6}\s+/.test(norm(lines[end])) &&  
!/^\s*---\s*$/.test(lines[end])  
) end++;  
  
const body = lines.slice(start + 1, end).join("\n").trim();  
return body.length ? body : null;  
}  
  
function renderBlock(title, content, fileLink) {  
dv.header(4, title);  
if (!content) {  
dv.paragraph("- (내용 없음)");  
return;  
}  
dv.paragraph(`${fileLink}`);  
dv.paragraph(content);  
}  
  
const today = pageDate(dv.current());  
if (!today) {  
dv.paragraph("현재 노트의 date(YY年 MM月 DD日) 또는 파일명에서 날짜를 파싱하지 못했습니다.");  
return;  
}  
  
const targets = [  
{ label: "일주일 전", dt: today.minus({ days: 7 }).startOf("day") },  
{ label: "한달 전", dt: today.minus({ months: 1 }).startOf("day") },  
{ label: "6개월 전", dt: today.minus({ months: 6 }).startOf("day") },  
{ label: "1년 전", dt: today.minus({ years: 1 }).startOf("day") },  
];  
  
const pagesDA = dv.pages(DAILY_FOLDER ? `"${DAILY_FOLDER}"` : "");  
const pages = (pagesDA?.array ? pagesDA.array() : Array.from(pagesDA ?? []));  
  
const map = new Map();  
for (const p of pages) {  
const d = pageDate(p);  
if (!d) continue;  
map.set(d.toISODate(), p);  
}  
  
function findPageByDate(dt) {  
return map.get(dt.toISODate()) ?? null;  
}  
  
let first = true;  
  
for (const t of targets) {  
const key = t.dt.toISODate();  
const p = findPageByDate(t.dt);  
  
if (!first) dv.paragraph("---");  
first = false;  
  
if (!p) {  
dv.header(4, t.label);  
dv.paragraph(`- (해당 날짜 노트 없음) target=${key}`);  
continue;  
}  
  
const section = await extractSectionText(p, SECTION_HEADING);  
renderBlock(`${t.label} (${key})`, section, dv.fileLink(p.file.path));  
}  
  
} catch (e) {  
dv.paragraph("DataviewJS 실행 오류:");  
dv.paragraph(String(e?.message ?? e));  
}
```
